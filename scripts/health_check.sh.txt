#!/bin/bash
# A simple script to capture a snapshot of system health.

REPORT_FILE="health_check_$(date +%Y-%m-%d_%H-%M-%S).txt"

echo "--- System Health Check ---" > $REPORT_FILE
echo "Date: $(date)" >> $REPORT_FILE
echo "" >> $REPORT_FILE

echo "--- Disk Usage ---" >> $REPORT_FILE
df -h >> $REPORT_FILE
echo "" >> $REPORT_FILE

echo "--- Memory Usage ---" >> $REPORT_FILE
free -h >> $REPORT_FILE
echo "" >> $REPORT_FILE

echo "--- Top 5 CPU Processes ---" >> $REPORT_FILE
ps aux --sort=-%cpu | head -n 6 >> $REPORT_FILE

echo "Report saved to $REPORT_FILE"
